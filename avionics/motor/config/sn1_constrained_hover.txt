# SN1 motor controller parameters for constrained flight.

# Open up limits to allow full power operation with some extra margin.
set_param all phase_current_limit 290
set_param all bus_current_limit 200

# Bus current command limits for the full stack of motors (i.e. tether current).
set_param all ibus_upper_limit 310
set_param all ibus_lower_limit -235

# Quadrature current command limits.
set_param all iq_upper_limit 225
set_param all iq_lower_limit -225

# Explicitly enable the stacking controller.
set_param all stacking_enable 1

# Prevent motors from turning off if they throw a warning.
set_param all shutdown_on_warning_enable 0

# Revert the motor status message rate to 10 Hz.
set_param all motor_status_message_period 0.1

# Early Yasa 2.3 motors are missing stator temperature sensors.
set_param SBO thermal_limits[kMotorThermalChannelStatorCore] 1e6
set_param SBO thermal_limits[kMotorThermalChannelStatorCoil] 1e6
set_param PTO thermal_limits[kMotorThermalChannelStatorCore] 1e6
set_param PTO thermal_limits[kMotorThermalChannelStatorCoil] 1e6

# PBO has bad capacitor sensor (pinched cable?).
set_param PBO thermal_limits[kMotorThermalChannelCapacitor] 1e6

# Yasa motors older than gen 2.3 have higher sin and cos scales.
set_param all angle_meas_upper_warn_lim 2300
